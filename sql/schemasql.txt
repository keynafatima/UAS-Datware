-- Dimensional model (templates)
-- Date dimension
CREATE TABLE IF NOT EXISTS dim_date (
  date_key serial PRIMARY KEY,
  date date UNIQUE,
  day int,
  month int,
  year int,
  day_name text
);

-- Site dimension
CREATE TABLE IF NOT EXISTS dim_site (
  site_sk serial PRIMARY KEY,
  site_id bigint,
  site_name text,
  region text,
  latitude numeric,
  longitude numeric,
  UNIQUE(site_id)
);

-- Equipment dimension (SCD2 template)
CREATE TABLE IF NOT EXISTS dim_equipment (
  equipment_sk serial PRIMARY KEY,
  equipment_name text,
  equipment_type text,
  manufacture text,
  model text,
  capacity numeric,
  purchase_date date,
  effective_from timestamp default now(),
  effective_to timestamp,
  is_current boolean default true
);

-- Material, employee dims (simple)
CREATE TABLE IF NOT EXISTS dim_material (
  material_sk serial PRIMARY KEY,
  material_id bigint,
  material_name text,
  material_type text,
  unit_of_measure text
  , UNIQUE(material_id)
);

CREATE TABLE IF NOT EXISTS dim_employee (
  employee_sk serial PRIMARY KEY,
  employee_id bigint,
  employee_name text,
  position text,
  department text
  , UNIQUE(employee_id)
);

-- Fact tables
CREATE TABLE IF NOT EXISTS fact_production (
  production_sk serial PRIMARY KEY,
  production_id bigint,
  date date,
  site_sk int REFERENCES dim_site(site_sk),
  material_sk int REFERENCES dim_material(material_sk),
  employee_sk int REFERENCES dim_employee(employee_sk),
  produced_volume numeric,
  unit_cost numeric
  , UNIQUE(production_id)
);

CREATE TABLE IF NOT EXISTS fact_transaction (
  transaction_sk serial PRIMARY KEY,
  id bigint,
  date date,
  site_sk int REFERENCES dim_site(site_sk),
  project_id bigint,
  account_id bigint,
  budgeted_cost numeric,
  actual_cost numeric,
  cost numeric
  , UNIQUE(id)
);
