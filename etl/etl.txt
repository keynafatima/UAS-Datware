import os
import pandas as pd
from sqlalchemy import create_engine, text


def _make_engine(db):
    url = f"postgresql+psycopg2://{db['user']}:{db['password']}@{db['host']}:{db['port']}/{db['db']}"
    return create_engine(url)


def profile_data(data_dir='/data'):
    files = [
        ('dataset_alat_berat_dw.csv', 'alat_berat'),
        ('dataset_production.csv', 'production'),
        ('dataset_transaksi.csv', 'transaksi'),
    ]
    for fname, label in files:
        path = os.path.join(data_dir, fname)
        if not os.path.exists(path):
            print(f"missing: {path}")
            continue
        print(f"--- {label} ({path}) ---")
        df = pd.read_csv(path, nrows=100)
        print(df.dtypes)
        print(df.head(3).to_dict(orient='records'))


def run_etl(db, data_dir='/data'):
    engine = _make_engine(db)
    # create schema
    sql_path = os.path.join(os.path.dirname(__file__), '..', 'sql', 'schema.sql')
    if os.path.exists(sql_path):
        with open(sql_path, 'r', encoding='utf-8') as f:
            ddl = f.read()
        with engine.begin() as conn:
            conn.execute(text(ddl))
        print('schema applied')

    # load staging using pandas.to_sql (replace)
    mapping = {
        'dataset_alat_berat_dw.csv': 'stg_alat_berat',
        'dataset_production.csv': 'stg_production',
        'dataset_transaksi.csv': 'stg_transaksi',
    }
    for fname, table in mapping.items():
        path = os.path.join(data_dir, fname)
        if not os.path.exists(path):
            print(f"skip missing {path}")
            continue
        print(f"loading {path} -> {table}")
        df = pd.read_csv(path, dtype=str)
        df.columns = [c.strip() for c in df.columns]
        # truncate existing staging table and append to preserve schema from schema.sql
        with engine.begin() as conn:
            conn.execute(text(f"TRUNCATE TABLE {table} RESTART IDENTITY;"))
        df.to_sql(table, engine, if_exists='append', index=False)
        print(f"loaded {len(df)} rows into {table}")

    # run transforms and loads
    transform_and_load(engine)
    print('ETL finished.')


def _dq_check_counts(engine):
    checks = {}
    with engine.connect() as conn:
        for t in ('stg_alat_berat', 'stg_production', 'stg_transaksi'):
            r = conn.execute(text(f"SELECT count(*) FROM {t}"))
            checks[t] = r.scalar()
    print('DQ counts:', checks)
    return checks


def transform_and_load(engine):
    # basic DQ
    counts = _dq_check_counts(engine)
    if all(v == 0 for v in counts.values()):
        print('no data in staging, aborting transforms')
        return

    with engine.begin() as conn:
        # upsert dim_site from stg_transaksi and stg_production
        # Insert site ids only when numeric to avoid type cast errors
        conn.execute(text(
            "INSERT INTO dim_site(site_id, site_name, region, latitude, longitude)"
            " SELECT DISTINCT (site_id::bigint), site_name, region, latitude, longitude"
            " FROM stg_transaksi"
            " WHERE CAST(site_id AS text) ~ '^\\\d+$'"
            " ON CONFLICT (site_id) DO UPDATE SET"
            " site_name=EXCLUDED.site_name, region=EXCLUDED.region, latitude=EXCLUDED.latitude, longitude=EXCLUDED.longitude;"
        ))
        conn.execute(text(
            "INSERT INTO dim_site(site_id, site_name, region, latitude, longitude)"
            " SELECT DISTINCT (site_id::bigint), site_name, region, latitude, longitude"
            " FROM stg_production"
            " WHERE CAST(site_id AS text) ~ '^\\\d+$'"
            " ON CONFLICT (site_id) DO UPDATE SET"
            " site_name=EXCLUDED.site_name, region=EXCLUDED.region, latitude=EXCLUDED.latitude, longitude=EXCLUDED.longitude;"
        ))

        # ensure dim_material has unique on material_id (best-effort)
        # upsert dim_material - only numeric material_id
        conn.execute(text(
            "INSERT INTO dim_material(material_id, material_name, material_type, unit_of_measure)"
            " SELECT DISTINCT (material_id::bigint), material_name, material_type, unit_of_measure"
            " FROM stg_production WHERE CAST(material_id AS text) ~ '^\\\d+$'"
            " ON CONFLICT (material_id) DO UPDATE SET"
            " material_name=EXCLUDED.material_name, material_type=EXCLUDED.material_type, unit_of_measure=EXCLUDED.unit_of_measure;"
        ))

        # ensure dim_employee unique
        # upsert dim_employee - only numeric employee_id
        conn.execute(text(
            "INSERT INTO dim_employee(employee_id, employee_name, position, department)"
            " SELECT DISTINCT (employee_id::bigint), employee_name, position, department"
            " FROM stg_production WHERE CAST(employee_id AS text) ~ '^\\\d+$'"
            " ON CONFLICT (employee_id) DO UPDATE SET"
            " employee_name=EXCLUDED.employee_name, position=EXCLUDED.position, department=EXCLUDED.department;"
        ))

        # SCD2 for equipment (dim_equipment) based on equipment_name + model
        rows = conn.execute(text(
            "SELECT DISTINCT equipment_name, equipment_type, manufacture, model, capacity, purchase_date"
            " FROM stg_alat_berat"
        )).fetchall()
        for r in rows:
            equipment_name, equipment_type, manufacture, model, capacity, purchase_date = r
            cur = conn.execute(text(
                "SELECT equipment_sk, equipment_name, equipment_type, manufacture, model, capacity"
                " FROM dim_equipment WHERE equipment_name = :name AND model = :model AND is_current = true"
            ), {'name': equipment_name, 'model': model}).fetchone()
            if cur is None:
                # insert new current
                conn.execute(text(
                    "INSERT INTO dim_equipment(equipment_name, equipment_type, manufacture, model, capacity, purchase_date, effective_from, is_current)"
                    " VALUES(:name,:etype,:manuf,:model,:cap,:pdate, now(), true)"
                ), {'name': equipment_name, 'etype': equipment_type, 'manuf': manufacture, 'model': model, 'cap': capacity, 'pdate': purchase_date})
            else:
                # compare attributes; if changed, expire and insert new
                row = cur._mapping if hasattr(cur, '_mapping') else None
                db_type = str(row['equipment_type']) if row is not None else str(cur[2])
                db_manufacture = str(row['manufacture']) if row is not None else str(cur[3])
                db_capacity = str(row['capacity']) if row is not None else str(cur[5])
                if (db_type != str(equipment_type)) or (db_manufacture != str(manufacture)) or (db_capacity != str(capacity)):
                    sk = row['equipment_sk'] if row is not None else cur[0]
                    conn.execute(text(
                        "UPDATE dim_equipment SET effective_to = now(), is_current = false WHERE equipment_sk = :sk"
                    ), {'sk': sk})
                    conn.execute(text(
                        "INSERT INTO dim_equipment(equipment_name, equipment_type, manufacture, model, capacity, purchase_date, effective_from, is_current)"
                        " VALUES(:name,:etype,:manuf,:model,:cap,:pdate, now(), true)"
                    ), {'name': equipment_name, 'etype': equipment_type, 'manuf': manufacture, 'model': model, 'cap': capacity, 'pdate': purchase_date})

        # Load facts - production
        # date parsing: handle YYYYMMDD or YYYY-MM-DD
        prod_sql = text("""
        INSERT INTO fact_production(production_id, date, site_sk, material_sk, employee_sk, produced_volume, unit_cost)
        SELECT p.production_id,
            (CASE WHEN length(trim(p.date))=8 THEN to_date(p.date,'YYYYMMDD') ELSE to_date(p.date,'YYYY-MM-DD') END) as dt,
            s.site_sk, m.material_sk, e.employee_sk, p.produced_volume, p.unit_cost
        FROM stg_production p
        JOIN dim_site s ON s.site_id::text = trim(CAST(p.site_id AS text))
        JOIN dim_material m ON m.material_id::text = trim(CAST(p.material_id AS text))
        JOIN dim_employee e ON e.employee_id::text = trim(CAST(p.employee_id AS text))
        ON CONFLICT DO NOTHING;
        """)
        conn.execute(prod_sql)

        # Load transaction facts
        trans_sql = text("""
        INSERT INTO fact_transaction(id, date, site_sk, project_id, account_id, budgeted_cost, actual_cost, cost)
        SELECT t.id,
            (CASE WHEN length(trim(t.date))=8 THEN to_date(t.date,'YYYYMMDD') ELSE to_date(t.date,'YYYY-MM-DD') END) as dt,
            s.site_sk, t.project_id, t.account_id, t.budgeted_cost, t.actual_cost, t.cost
        FROM stg_transaksi t
        LEFT JOIN dim_site s ON s.site_id::text = trim(CAST(t.site_id AS text))
        ON CONFLICT DO NOTHING;
        """)
        conn.execute(trans_sql)

    print('transforms and loads applied')
